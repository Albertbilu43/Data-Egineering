	Types: Permanent, Temporary, Transcient, External

	Permanent: Default table type. They persist indefinitely until dropped and include full Snowflake Time Travel (up to 90 days) and 
		   a 7-day Fail-safe period for disaster recovery.
	Temporary: Session-specific, automatically purged when the session ends. They do not support Fail-safe and offer only 
                   a maximum of 1 day of Time Travel. They are useful for intermediate ETL data or quick aggregations.
	Transient: Persistent until explicitly dropped, making them available to multiple users and sessions. They lack Fail-safe protection and limit Time Travel 
		   to 1 day, which helps reduce storage costs for staging data.
	External:  Read-only tables that allow you to query data files stored in external cloud storage (like AWS S3, Azure Blob, or Google Cloud Storage) without 
		   loading or moving them into Snowflake's native storage
	Dynamic:   Automatically update based on a SQL query definition and a target data freshness lag. Great for declarative, continuous data transformations 
		   instead of complex task scheduling.
	Hybrid:    High-performance tables tailored for low-latency operational (OLTP) read/write transactional queries, while simultaneously integrating 
                   with Snowflake's analytical storage engine.
	Apache 
        Iceberg:  Built on the open-source Iceberg table format. These allow external storage platforms to own the data while Snowflake processes the compute, 
		  reducing vendor lock-in.
	Event:    Specifically for log tracking, tracing, and operational telemetry. They help monitor, debug, and log custom execution messages from 
                  stored procedures and user-defined functions (UDFs)

CREATE TABLE snow_db.schema1.table1 AS (

SELECT
	DATE ,
	CAMPAIGN_ID ,
	CAMPAIGN_NAME ,
	AD_SET_NAME AS AD_GROUP_NAME,
	IMPRESSIONS ,
	CLICKS ,
	SPEND ,
	CONVERSIONS ,
    NULL AS "conversion_value" ,
	VIDEO_VIEWS ,
	ENGAGEMENT_RATE,
	REACH ,
	(CLICKS/IMPRESSIONS * 100.0) AS "CTR"
  FROM DB_SOCIAL.PUBLIC.FACEBOOK_ADS  
  
  UNION
  SELECT	DATE ,
	CAMPAIGN_ID ,
	CAMPAIGN_NAME ,
	AD_GROUP_NAME,
	IMPRESSIONS ,
	CLICKS ,
	COST AS SPEND ,
	CONVERSIONS ,
    conversion_value ,
	NULL AS "VIDEO_VIEWS" ,
	(CLICKS/IMPRESSIONS) AS ENGAGEMENT_RATE,
	NULL AS REACH ,
	CTR
  FROM DB_SOCIAL.PUBLIC.GOOGLE_ADS 
    
  UNION
    
  SELECT 	DATE ,
	CAMPAIGN_ID ,
	CAMPAIGN_NAME ,
	ADGROUP_NAME AS AD_GROUP_NAME,
	IMPRESSIONS ,
	CLICKS ,
	COST AS SPEND ,
	CONVERSIONS ,
    NULL AS "conversion_value",
	VIDEO_VIEWS ,
	(CLICKS/IMPRESSIONS) AS ENGAGEMENT_RATE,
	NULL AS REACH ,
	(CLICKS/IMPRESSIONS * 100.0) AS "CTR"
  FROM DB_SOCIAL.PUBLIC.TIKTOK_ADS 
);

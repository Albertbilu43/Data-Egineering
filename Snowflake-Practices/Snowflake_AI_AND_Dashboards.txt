SELECT DISTINCT  CASE (campaign_id) 
                     WHEN  'fb_1001' Then 'FACEBOOK'
                     WHEN  'fb_1002' Then 'FACEBOOK'
                     WHEN  'fb_1003' Then 'FACEBOOK'
                     WHEN  'fb_1004' Then 'FACEBOOK'
                     WHEN  'g_5001'  Then 'GOOGLE'
                     WHEN  'g_5002'  Then 'GOOGLE'
                     WHEN  'g_5003'  Then 'GOOGLE'
                     WHEN  'g_5004'  Then 'GOOGLE'   
                     WHEN  'tt_8001' Then 'TIKTOK'
                     WHEN  'tt_8002' Then 'TIKTOK'
                     WHEN  'tt_8003' Then 'TIKTOK'
                     WHEN  'tt_8004' Then 'TIKTOK'                      
                     ELSE campaign_id

                     END AS SOC_NETWORK
                     
              ,SUM(SPEND)	    AS SPEND
              ,SUM(impressions) AS impressions
              ,SUM(clicks)	    AS clicks
              ,SUM(conversions)	AS conversions
              ,SUM(video_views) AS video_views
              ,SUM(reach) AS reach
   
FROM "DB_SOCIAL"."PUBLIC"."MERGED_SOC_ADS"
--WHERE campaign_id = 'fb_1001'
GROUP BY    CASE (campaign_id) 
                     WHEN  'fb_1001' Then 'FACEBOOK'
                     WHEN  'fb_1002' Then 'FACEBOOK'
                     WHEN  'fb_1003' Then 'FACEBOOK'
                     WHEN  'fb_1004' Then 'FACEBOOK'
                     WHEN  'g_5001'  Then 'GOOGLE'
                     WHEN  'g_5002'  Then 'GOOGLE'
                     WHEN  'g_5003'  Then 'GOOGLE'
                     WHEN  'g_5004'  Then 'GOOGLE'

                     WHEN  'tt_8003' Then 'TIKTOK'
                     WHEN  'tt_8004' Then 'TIKTOK'

                     END
 ORDER BY SPEND DESC
